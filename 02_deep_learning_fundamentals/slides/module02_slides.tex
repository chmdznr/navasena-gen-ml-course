\begin{frame}{Apa itu Neural Network?}
  \framesubtitle{Meniru cara otak memproses informasi}

    \begin{column}{0.46\textwidth}

        \footnotesize

        \item \textbf{Flatten}: 28$\times$28 jadi 784 angka

model = tf.keras.Sequential([
  layers.Flatten(input_shape=(28,28)),
  layers.Dense(128, activation='relu'),
  layers.Dense(64,  activation='relu'),
  layers.Dense(10,  activation='softmax')
])
      \end{lstlisting}
    \end{column}
  \end{columns}

  epochs=20, batch_size=32,

% Slide 7 — Kenapa Aktivasi
\begin{frame}[fragile]{Kenapa Perlu Aktivasi?}
  \framesubtitle{Tanpa non-linearity, jaringan tetap satu garis lurus}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \footnotesize
        \item Tanpa \textbf{activation}, NN cuma jumlah \& kali
        \item Banyak layer linear ditumpuk $=$ tetap 1 fungsi linear
        \item \textbf{Activation} tambah \textbf{non-linearity} $\to$ kenali pola berkelok
        \item Inti pengenalan gambar, teks, dan suara
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{lstlisting}[language=Python, basicstyle=\ttfamily\scriptsize\color{nvwhite}]
layer2     = W2 @ (W1 @ x)
W_gabungan = W2 @ W1
satu_layer = W_gabungan @ x
# layer2 == satu_layer (sama!)
      \end{lstlisting}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{beamercolorbox}[rounded=true,shadow=false,sep=4pt]{block body}
    \centering\small\color{white}
    Tanpa activation, 100 layer $=$ 1 layer linear belaka.
  \end{beamercolorbox}

        \item \textbf{Tanh}: -1--1, zero-centered, stabil
        \item \textbf{LeakyReLU}: gradient kecil di sisi negatif
        \item Risiko: \textbf{vanishing gradient}, \textbf{dead neuron}

% Slide 13 — Membangun CNN
\begin{frame}[fragile]{Membangun CNN dari Nol}
  \framesubtitle{Conv2D + MaxPooling2D + Dense pada CIFAR-10}
  \begin{columns}[T]
    \begin{column}{0.44\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{CIFAR-10}: 60K gambar warna 32$\times$32, 10 kelas
        \item Blok \texttt{Conv2D} (\textbf{ReLU}) $\to$ \texttt{MaxPooling2D}
        \item Tiap pooling: dimensi 32$\to$16$\to$8$\to$4, filter bertambah
        \item \texttt{Flatten} $\to$ \texttt{Dense} $\to$ \texttt{Dropout} $\to$ \texttt{Dense(10)}
      \end{itemize}
    \end{column}
    \begin{column}{0.54\textwidth}
      \begin{lstlisting}[language=Python, basicstyle=\ttfamily\tiny\color{nvwhite}]
model = Sequential([
  Conv2D(32,(3,3),activation='relu',padding='same',
         input_shape=(32,32,3)),
  MaxPooling2D(),               # 32->16
  Conv2D(64,(3,3),activation='relu',padding='same'),
  MaxPooling2D(),               # 16->8
  Conv2D(128,(3,3),activation='relu',padding='same'),
  MaxPooling2D(),               # 8->4
  Flatten(), Dense(128,'relu'),
  Dropout(0.5), Dense(10,'softmax')
])
      \end{lstlisting}
    \end{column}
  \end{columns}
\end{frame}

% Slide 15 — Transfer Learning
\begin{frame}[fragile]{Transfer Learning}
  \framesubtitle{Pakai ResNet50 yang sudah pintar}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \begin{itemize}
        \footnotesize
        \item Analogi: fasih B.Indonesia $\to$ belajar B.Melayu lebih cepat
        \item \textbf{ResNet50} pretrained di ImageNet (14 juta gambar)
        \item \textbf{Freeze} layer bawah: \texttt{trainable=False}
        \item Ganti \textbf{head}: pooling $\to$ Dense $\to$ Dense(10)
      \end{itemize}
    \end{column}
    \begin{column}{0.52\textwidth}
      \begin{lstlisting}[language=Python, basicstyle=\ttfamily\scriptsize\color{nvwhite}]
base = ResNet50(weights='imagenet',
  include_top=False,
  input_shape=(32,32,3))
base.trainable = False   # freeze
model = Sequential([base,
  GlobalAveragePooling2D(),
  Dense(128,'relu'),
  Dense(10,'softmax')])
      \end{lstlisting}
    \end{column}
  \end{columns}
\end{frame}

% Slide 17 — Data Berurutan & RNN
\begin{frame}{Data Berurutan \& RNN}
  \framesubtitle{Neural network yang punya 'memori'}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{Sequential data}: urutan PENTING (teks, saham, cuaca)
        \item "Saya suka kucing" $\neq$ "Kucing suka saya"
        \item Dense/CNN proses tiap input independen (tanpa memori)
        \item \textbf{RNN} ingat langkah sebelumnya via \textbf{hidden state} ($h$)
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{center}
      \begin{tikzpicture}[scale=0.7, every node/.style={transform shape},

        \draw[->,thick,color=nvgreen] (h1) -- (h2);
        \draw[->,thick,color=nvgreen] (h2) -- (h3);
        \foreach \x/\n in {0/x_1,2/x_2,4/x_3} {
          \node[color=nvgray, font=\small] (\n) at (\x,-1.6) {$\n$};

        }

      \end{tikzpicture}
      \\[0.2em]
      {\tiny\color{nvgray}hidden state diteruskan antar timestep}
      \end{center}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}\textcolor{nvlightgreen}{\small RNN memproses input baru $+$ memori dari langkah sebelumnya (hidden state).}\end{center}
\end{frame}

% Slide 19 — Masalah RNN -> LSTM
\begin{frame}{Masalah RNN $\to$ LSTM}
  \framesubtitle{Mengatasi 'lupa' konteks panjang}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{Vanishing gradient}: urutan panjang $\to$ lupa info awal
        \item Seperti bisik berantai: pesan rusak di ujung
        \item \textbf{LSTM} punya \textbf{cell state} (memori jangka panjang)
        \item \textbf{Gates}: forget (hapus), input (tulis), output (baca)
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{center}
      \renewcommand{\arraystretch}{1.25}
      {\scriptsize
      \setlength{\tabcolsep}{3pt}
      \begin{tabular}{@{}p{0.38\linewidth}p{0.28\linewidth}p{0.28\linewidth}@{}}
        \toprule
        \rowcolor{nvcard!50}
        \textbf{Aspek} & \textbf{SimpleRNN} & \textbf{LSTM} \\
        \midrule
        Memori & Pendek & Panjang \\
        Parameter & Sedikit & $\sim$4$\times$ lebih \\
        Vanishing grad. & Sering & Teratasi \\
        Kecepatan & Cepat & Lebih lambat \\
        \bottomrule
      \end{tabular}%
      }
      \end{center}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}\textcolor{nvlightgreen}{\small LSTM punya 3 gates $+$ cell state agar info penting bertahan lewat banyak timestep.}\end{center}
\end{frame}

\acttitle{5}{Akselerasi GPU NVIDIA}{Kenapa deep learning butuh GPU}

% Slide 22 — Kenapa GPU
\begin{frame}{Kenapa DL Butuh GPU?}
  \framesubtitle{CPU vs GPU: serial vs paralel}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{CPU}: sedikit core kuat, kerja \textbf{serial}
        \item \textbf{GPU}: ribuan core sederhana, kerja \textbf{paralel}
        \item Inti NN $=$ \textbf{matrix multiplication}, sangat paralel
      \end{itemize}
    \end{column}
    \begin{column}{0.52\textwidth}
      \begin{center}
      \renewcommand{\arraystretch}{1.3}
      {\scriptsize
      \setlength{\tabcolsep}{3pt}
      \begin{tabular}{@{}p{0.30\linewidth}p{0.30\linewidth}p{0.34\linewidth}@{}}
        \toprule
        \rowcolor{nvcard!50}
        \textbf{Aspek} & \textbf{CPU} & \textbf{GPU NVIDIA} \\
        \midrule
        Core & 4--16 (kuat) & Ribuan (sederhana) \\
        Cocok & Tugas berurutan & Tugas paralel \\
        Deep Learning & Lambat & Cepat \\
        Contoh & i7, Ryzen & T4, A100, H100 \\
        \bottomrule
      \end{tabular}%
      }
      \end{center}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}\textcolor{nvlightgreen}{\small 1 guru jenius (CPU) vs 1.000 kalkulator serentak (GPU): DL pilih yang serentak.}\end{center}
\end{frame}

\begin{frame}[fragile]{Mixed Precision Training}
  \framesubtitle{Campur FP16 dan FP32}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{float16} (cepat): forward \& backward pass
        \item \textbf{float32}: bobot master \& akumulasi \textbf{gradient}
        \item \textbf{Tensor Cores} NVIDIA percepat operasi float16
        \item Hemat memori $\sim$50\%, akurasi nyaris tetap
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{lstlisting}[language=Python, basicstyle=\ttfamily\scriptsize\color{nvwhite}]
from tensorflow.keras import \
     mixed_precision
mixed_precision.set_global_policy(
     'mixed_float16')
# output tetap float32
Dense(10, 'softmax', dtype='float32')
      \end{lstlisting}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}\textcolor{nvlightgreen}{\small Lebih cepat $+$ hemat memori; dipakai melatih GPT, Stable Diffusion, Gemini.}\end{center}
\end{frame}

\acttitle{6}{Generative AI}{Saat neural network mencipta, bukan hanya mengenali}

\begin{frame}{Diskriminatif vs Generatif}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\color{nvgreen}\textbf{Diskriminatif (Act 1--5)}}
      \begin{itemize}
        \item Input data $\to$ output \textbf{label/angka}
        \item ``Gambar ini kucing atau anjing?''
        \item Belajar \textbf{batas} antar kelas
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      {\color{nvorange}\textbf{Generatif (Act 6)}}
      \begin{itemize}
        \item Noise/konteks $\to$ output \textbf{data baru}
        \item ``Buatkan gambar digit yang meyakinkan''
        \item Belajar \textbf{distribusi} data itu sendiri
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.5em}
  \begin{center}
    {\color{nvgray}\small Fondasi ChatGPT \& Stable Diffusion --- dan jembatan kita menuju Module 04 (LLM)}
  \end{center}
\end{frame}

\begin{frame}{Autoencoder \& Latent Space}
  \begin{itemize}
    \item Encoder memampatkan 784 pixel $\to$ \textbf{32 angka} (latent); decoder merekonstruksi
    \item Decoder pada dasarnya = \textbf{generator}: vektor angka $\to$ gambar
    \item Latent space menyimpan \textbf{struktur}: titik di antara dua digit juga valid
  \end{itemize}
  \begin{center}
    \includegraphics[width=0.85\linewidth]{figures/latent_interp.pdf}
  \end{center}
  {\color{nvgray}\small Tapi: sampling \emph{acak} dari latent space AE hasilnya kabur --- AE tidak dilatih untuk mencipta. Kita butuh GAN.}
\end{frame}

\begin{frame}{GAN: Generator vs Discriminator}
  \begin{center}
    \includegraphics[width=0.92\linewidth]{figures/gan_architecture.png}
  \end{center}
  \vspace{0.3em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item \textbf{Generator} = pemalsu: noise $\to$ gambar palsu
        \item \textbf{Discriminator} = kurator: asli atau palsu?
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item Dilatih \textbf{bersamaan} dalam satu langkah --- saling memaksa jadi lebih baik
        \item DCGAN: generator \texttt{Conv2DTranspose}, discriminator \texttt{Conv2D}
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{DCGAN: Digit Lahir dari Noise}
  \begin{center}
    \includegraphics[width=0.8\linewidth]{figures/gan_grid.pdf}
  \end{center}
  \begin{itemize}
    \item Noise input \textbf{sama} di tiap baris --- yang berubah hanya kemampuan generator
    \item Epoch 1: noise berbintik $\to$ epoch 15: mayoritas berbentuk digit
    \item Catatan jujur: GAN rewel ---  \emph{mode collapse} (variasi macet) sering terjadi
  \end{itemize}
\end{frame}

\begin{frame}{Peta Generative AI Modern}
  \setlength{\tabcolsep}{3pt}
  {\small
  \begin{tabular}{p{0.2\linewidth}p{0.42\linewidth}p{0.28\linewidth}}
    {\color{nvgreen}\textbf{Keluarga}} & {\color{nvgreen}\textbf{Ide inti}} & {\color{nvgreen}\textbf{Contoh}} \\
    \hline
    VAE & latent space diatur supaya bisa di-sampling & generator wajah awal \\
    GAN & generator vs discriminator berlomba & StyleGAN, deepfake \\
    Diffusion & generate = denoise bertahap dari noise murni & Stable Diffusion, DALL-E \\
    Autoregressive & prediksi token berikutnya, berulang & GPT, ChatGPT, Llama \\
  \end{tabular}}
  \vspace{0.6em}
  \begin{itemize}
    \item Benang merah: semua \textbf{belajar distribusi data}, lalu sampling darinya
    \item ChatGPT $\approx$ DCGAN kamu: sama-sama generator --- bedanya token, bukan pixel
  \end{itemize}
  \vspace{0.3em}
  {\color{nvlightgreen}\small $\Rightarrow$ Module 04: hands-on dengan generator teks (LLM)}
\end{frame}
