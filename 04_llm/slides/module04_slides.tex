% Slide — Embedding = makna jadi angka (HOW IT UNDERSTANDS, req 2)
\begin{frame}{Langkah 2 --- Embedding: Mengubah Makna jadi Angka}
  \framesubtitle{Inilah cara LLM ``memahami'' bahasa}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{itemize}
        \footnotesize
        \item Tiap token diubah jadi \textbf{deret angka} (disebut \emph{vektor})
        \item Aturannya: \textbf{makna mirip $\to$ angka mirip}; makna jauh $\to$ angka jauh
        \item ``raja'' dekat ``ratu''; ``kucing'' jauh dari ``traktor''
        \item Jadi ``memahami'' di sini $=$ menaruh tiap kata di \emph{peta angka} sesuai maknanya
        \item Model menyusun peta ini sendiri dari \textbf{pola} dalam teks yang sangat banyak
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \centering

        \node[nvgray,font=\tiny] at (1.7,-0.45) {peta makna (disederhanakan)};
      \end{tikzpicture}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{beamercolorbox}[rounded=true,shadow=false,sep=4pt]{block body}
    \small\color{white}
    LLM ``memahami'' $=$ mengenali \textbf{pola} lalu menaruh makna sebagai \textbf{angka}. Ini statistik dan pola --- bukan kesadaran atau perasaan seperti manusia.
  \end{beamercolorbox}
\end{frame}

% Slide — Attention intuitif
\begin{frame}{Langkah 3 --- Attention: Memakai Konteks}
  \framesubtitle{Model belajar kata mana yang saling berhubungan}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{Attention} $=$ model ``memperhatikan'' kata yang paling relevan
        \item Saat mengolah satu kata, ia menimbang semua kata lain di kalimat
        \item Contoh: model belajar \textbf{``itu''} merujuk ke \textbf{``buku''}, bukan ``meja''
        \item Inilah cara LLM memakai \textbf{konteks} --- kunci memahami kalimat panjang
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{center}

    Attention membuat model fokus pada kata yang benar-benar relevan, bukan menganggap semua kata sama penting.
  \end{beamercolorbox}
\end{frame}

% Slide — Q, K, V (req 3, group-discussion analogy) — analogi & roles
\begin{frame}{Q, K, V --- Analogi Diskusi Kelompok}
  \framesubtitle{Mesin di dalam attention, tanpa rumus}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      \footnotesize
      Bayangkan satu kelas sedang diskusi. \textbf{Tiap kata adalah seorang murid}:
      \begin{itemize}
        \item \textcolor{nvorange}{\textbf{Query (Q)}} $=$ \emph{pertanyaanku}: ``murid mana yang relevan untukku?''
        \item \textcolor{nvblue}{\textbf{Key (K)}} $=$ \emph{name-tag keahlian} tiap murid: ``aku tahu soal X''
        \item \textcolor{nvgreen}{\textbf{Value (V)}} $=$ \emph{isi informasi} yang dibawa tiap murid
      \end{itemize}
      \vspace{0.3em}
      \textbf{Cara kerjanya}: cocokkan \textbf{Q}-ku ke semua \textbf{K}. Murid yang paling cocok dapat \emph{bobot} besar, lalu aku ambil \emph{campuran-berbobot} dari \textbf{V} mereka.
    \end{column}
    \begin{column}{0.42\textwidth}
      \centering
      \resizebox{\linewidth}{!}{%
      \begin{tikzpicture}
        \node[draw=nvorange, fill=nvcard, rounded corners=4pt, text=nvorange,
              font=\scriptsize\bfseries, minimum width=1.8cm] (q) at (0,2.2) {Q: tanyaku};
        \node[draw=nvblue, fill=nvcard, rounded corners=4pt, text=nvblue,
              font=\scriptsize, minimum width=1.3cm] (k1) at (-1.7,0.8) {K\textsubscript{1}};
        \node[draw=nvblue, fill=nvcard, rounded corners=4pt, text=nvblue,
              font=\scriptsize, minimum width=1.3cm] (k2) at (0,0.8) {K\textsubscript{2}};
        \node[draw=nvblue, fill=nvcard, rounded corners=4pt, text=nvblue,
              font=\scriptsize, minimum width=1.3cm] (k3) at (1.7,0.8) {K\textsubscript{3}};
        \node[draw=nvgreen, fill=nvcard, rounded corners=4pt, text=nvgreen,
              font=\scriptsize, minimum width=2.4cm, align=center] (v) at (0,-0.8) {campur V\\(berbobot)};
        \draw[->,nvgray] (q)--(k1);
        \draw[->,nvgreen,thick] (q)--(k2);
        \draw[->,nvgray] (q)--(k3);
        \draw[->,nvgreen,thick] (k2)--(v);
      \end{tikzpicture}%
      }
      \\[0.2em]
      {\tiny\color{nvgray}garis tebal $=$ bobot besar (paling cocok)}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}
    \textcolor{nvlightgreen}{\small Q $=$ pertanyaan $\cdot$ K $=$ label keahlian $\cdot$ V $=$ isi info. Attention $=$ ambil campuran V yang paling cocok.}
  \end{center}
\end{frame}

% Slide — Output projection O (req 3, part O)
\begin{frame}{O --- Merangkum Hasil Diskusi}
  \framesubtitle{Output projection: langkah terakhir attention}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \footnotesize
      Setelah tiap kata mengambil campuran info dari kata lain, hasilnya masih ``mentah''. Perlu satu langkah terakhir:
      \begin{itemize}
        \item \textcolor{nvlightgreen}{\textbf{Output projection (O)}} $=$ \emph{merangkum} info yang terkumpul jadi \textbf{catatan baru} untuk kata itu
        \item Analogi: setelah bertanya ke teman-teman sekelas, kamu \textbf{menulis ringkasan} di catatanmu sendiri
        \item Catatan baru inilah yang dipakai lapisan berikutnya
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \centering
      \resizebox{\linewidth}{!}{%
      \begin{tikzpicture}[node distance=6mm,
          b/.style={draw, rounded corners=4pt, fill=nvcard,
                    font=\scriptsize, minimum width=3.0cm, minimum height=0.9cm, align=center}]
        \node[b, draw=nvgreen, text=nvgreen] (v) {campuran V\\\scriptsize(info terkumpul)};
        \node[b, draw=nvlightgreen, text=nvlightgreen, below=of v] (o) {O: rangkuman\\\scriptsize(catatan baru kata)};
        \draw[->,nvlightgreen,thick] (v)--(o) node[midway,right,nvlightgreen,font=\tiny]{rangkum};
      \end{tikzpicture}%
      }
    \end{column}
  \end{columns}
  \vspace{0.3em}
  \begin{beamercolorbox}[rounded=true,shadow=false,sep=4pt]{block body}
    \centering\small\color{white}
    Urutan lengkap: \textbf{Q} tanya $\to$ cocokkan ke \textbf{K} $\to$ ambil campuran \textbf{V} $\to$ \textbf{O} merangkum jadi catatan baru.
  \end{beamercolorbox}
\end{frame}

% Slide — Multi-head attention (one idea)
\begin{frame}{Langkah 4a --- Multi-Head: Banyak Diskusi Paralel}
  \framesubtitle{Beberapa attention sekaligus, lalu digabung}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \footnotesize
      \begin{itemize}

    \textcolor{nvlightgreen}{\small fp16 nyaris menyentuh langit-langit T4; 4-bit menyusutkannya jadi $\sim$1/3.}
  \end{center}
\end{frame}

% Slide — 4-bit NF4 quantization code
\begin{frame}[fragile]{4-bit NF4 Quantization}
  \framesubtitle{BitsAndBytesConfig --- dan jebakan T4}
  \begin{columns}[T]
    \begin{column}{0.42\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{NF4} $=$ format 4-bit yang pintar untuk menyimpan bobot model
        \item \texttt{compute\_dtype=float16} \textbf{WAJIB} di T4
        \item \textcolor{nvred}{\textbf{bf16 membuat crash}} di T4
        \item Hasil: 7B muat di $\sim$\textbf{5,5GB} memori
      \end{itemize}
    \end{column}
    \begin{column}{0.56\textwidth}
      \begin{lstlisting}[language=Python]
from transformers import BitsAndBytesConfig
import torch

bnb = BitsAndBytesConfig(
    load_in_4bit=True,
    bnb_4bit_quant_type="nf4",
    bnb_4bit_compute_dtype=torch.float16,  # bukan bf16!
)
model = AutoModelForCausalLM.from_pretrained(
    "mistralai/Mistral-7B-Instruct-v0.3",
    quantization_config=bnb, device_map="auto")
      \end{lstlisting}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{beamercolorbox}[rounded=true,shadow=false,sep=4pt]{block body}
    \centering\small\color{white}
    Ingat: di T4, \texttt{compute\_dtype} harus \texttt{float16}. \texttt{bfloat16} membuat kernel crash.
  \end{beamercolorbox}
\end{frame}

  \begin{columns}[T]

% Slide — masalah: kenapa tak latih ulang semuanya
\begin{frame}{Masalah: Melatih Ulang Model Raksasa Itu Mahal}
  \framesubtitle{Miliaran ``kenop'' --- terlalu berat untuk diubah semua}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \footnotesize
      \begin{itemize}
        \item Model raksasa punya \textbf{miliaran ``kenop''} (angka bobot) yang menentukan perilakunya
        \item Untuk mengajarinya hal baru, kita ingin \emph{menyetel} kenop-kenop itu
        \item Tapi menyetel \textbf{semua} kenop butuh GPU sangat mahal \& waktu sangat lama
        \item Pertanyaannya: bisakah kita ajari hal baru \textbf{tanpa} menyentuh semuanya?
      \end{itemize}
      \vspace{0.3em}
      \textcolor{nvlightgreen}{\textbf{Jawabannya: LoRA.}}
    \end{column}
    \begin{column}{0.44\textwidth}
      \centering
      \resizebox{0.9\linewidth}{!}{%

              font=\small, minimum width=3.4cm, minimum height=2.8cm, align=center]
              (book) {Model raksasa\\[2pt]\scriptsize miliaran kenop\\\scriptsize (mahal diubah semua)};
      \end{tikzpicture}%
      }
    \end{column}
  \end{columns}
\end{frame}

% Slide — LoRA sticky-note analogy (req 4 part 1)
\begin{frame}{LoRA --- Tempel Sticky-Note, Jangan Tulis Ulang Buku}
  \framesubtitle{Bekukan model raksasa, latih catatan kecil saja}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \footnotesize
      \textbf{LoRA} $=$ \emph{Low-Rank Adaptation}. Idenya:
      \begin{itemize}
        \item \textcolor{nvblue}{\textbf{Bekukan}} model raksasa --- jangan sentuh kenopnya sama sekali
        \item Tambahkan \textbf{matriks kecil} (``catatan tepi'') yang \emph{dilatih}
        \item \textbf{Analogi}: daripada menulis ulang seluruh buku teks, cukup tempel \textbf{sticky-note} kecil berisi koreksi/tambahan
        \item Yang dilatih hanya $\sim$\textbf{1\%} dari seluruh parameter
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \centering
      \resizebox{\linewidth}{!}{%
      \begin{tikzpicture}
        \node[draw=nvgray, fill=nvcard, rounded corners=4pt, text=nvgray,
              font=\small, minimum width=3.0cm, minimum height=2.6cm, align=center]
              (book) {Model raksasa\\(BEKU)\\[2pt]\scriptsize miliaran kenop};
        \node[draw=nvorange, fill=nvdarkcard, rounded corners=2pt, text=nvorange,
              font=\scriptsize\bfseries, minimum width=1.5cm, minimum height=0.8cm,
              rotate=8] at (1.4,1.0) {sticky-note};
        \node[draw=nvgreen, fill=nvdarkcard, rounded corners=2pt, text=nvgreen,
              font=\scriptsize\bfseries, minimum width=1.5cm, minimum height=0.8cm,
              rotate=-6] at (1.5,-0.9) {LoRA $\sim$1\%};
      \end{tikzpicture}%
      }
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}
    \textcolor{nvlightgreen}{\small Latih hanya sticky-note kecilnya $\to$ cepat, murah, mudah disimpan \& dibagi.}
  \end{center}
\end{frame}

% Slide — QLoRA (req 4 part 2)
\begin{frame}{QLoRA --- Buku Edisi Saku 4-bit + Sticky-Note}
  \framesubtitle{Gabungkan 4-bit dengan LoRA agar muat di T4}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \footnotesize
      \textbf{QLoRA} $=$ \textbf{Quantized} LoRA. Dua langkah:
      \begin{enumerate}
        \item \textcolor{nvblue}{\textbf{Kecilkan dulu}} model ke \textbf{4-bit} (quantization) agar muat di GPU kecil
        \item \textcolor{nvgreen}{\textbf{Baru tempel}} sticky-note \textbf{LoRA} di atasnya
      \end{enumerate}
      \vspace{0.2em}
      \textbf{Analogi}: buku tebal diubah jadi \textbf{edisi saku 4-bit} (ringan dibawa), lalu ditempeli \textbf{sticky-note} LoRA.
      \begin{itemize}
        \item Model 4-bit tetap dibekukan; hanya sticky-note LoRA yang dilatih
        \item Inilah kenapa fine-tuning bisa jalan di \textbf{T4 gratis} dalam $\sim$2 menit
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \centering
      \resizebox{\linewidth}{!}{%
      \begin{tikzpicture}[node distance=4mm,
          b/.style={draw, rounded corners=4pt, fill=nvcard, text=nvwhite,
                    font=\scriptsize, minimum width=3.2cm, minimum height=0.9cm, align=center}]
        \node[b, draw=nvgray] (full) {Model penuh\\\scriptsize berat, tak muat};
        \node[b, draw=nvblue, below=of full] (q) {Edisi saku 4-bit\\\scriptsize quantization};
        \node[b, draw=nvgreen, below=of q] (l) {+ sticky-note LoRA\\\scriptsize dilatih $\sim$1\%};
        \draw[->,nvblue,thick] (full)--(q) node[midway,right,nvblue,font=\tiny]{kecilkan};
        \draw[->,nvgreen,thick] (q)--(l) node[midway,right,nvgreen,font=\tiny]{tempel};
      \end{tikzpicture}%
      }
    \end{column}
  \end{columns}
\end{frame}

% Slide — LoraConfig + Trainer
\begin{frame}[fragile]{QLoRA dalam Kode}
  \framesubtitle{Notebook 03 --- ajari Qwen2.5-1.5B fakta-fakta Navasena}
  \begin{columns}[T]
    \begin{column}{0.42\textwidth}
      \begin{itemize}
        \footnotesize
        \item Model 4-bit $+$ \textbf{LoraConfig}: hanya $\sim$1,2\% parameter yang dilatih
        \item Dilatih lewat \texttt{Trainer} dalam $\sim$2 menit
        \item \textbf{Bandingkan} sebelum vs sesudah: \texttt{model.disable\_adapter()}

    task_type="CAUSAL_LM",
)
model = get_peft_model(base_4bit, lora)
model.print_trainable_parameters()  # ~1.2%

    base_answer = model.generate(**inputs)
      \end{lstlisting}
    \end{column}
  \end{columns}

\begin{frame}{Apakah Bedanya Nyata?}
