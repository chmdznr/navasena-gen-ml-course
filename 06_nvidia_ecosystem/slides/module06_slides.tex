\acttitle{2}{GPU, Precision \& Optimasi}{Kenapa GPU cepat, dan tiga tuas memperkecil model}

% ── Kenapa GPU ────────────────────────────────────────────
\begin{frame}{Kenapa GPU untuk LLM?}
  \framesubtitle{Inti tiap layer transformer adalah perkalian matriks (matmul)}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{itemize}\footnotesize
        \item Matmul itu \textbf{embarrassingly parallel}: tiap elemen hasil dihitung mandiri
        \item \textbf{CPU}: belasan core ``pintar'' --- logika bercabang, berurutan
        \item \textbf{GPU}: ribuan core sederhana --- operasi \textbf{sama} pada \textbf{banyak data} serentak
        \item \textbf{Tensor Core} (sejak Volta): hardware khusus perkalian-akumulasi matriks kecil
        \item Syaratnya: data \textbf{precision rendah} (FP16/FP8), bukan FP32
      \end{itemize}
    \end{column}
    \begin{column}{0.43\textwidth}
      \begin{center}
      \begin{tikzpicture}[scale=0.85, every node/.style={transform shape}]
        \node[draw=nvgray, fill=nvcard, rounded corners=4pt, text=white, font=\scriptsize,
              minimum width=2.0cm, minimum height=1.4cm, align=center] (cpu) {CPU\\\tiny\color{nvgray}belasan core};
        \node[draw=nvgreen, fill=nvcard, rounded corners=4pt, text=white, font=\scriptsize,
              minimum width=2.4cm, minimum height=1.4cm, align=center, right=0.6cm of cpu] (gpu) {GPU\\\tiny\color{nvlightgreen}ribuan core\\+ Tensor Core};
        \node[text=nvgray, font=\tiny, below=0.15cm of cpu] {beberapa koki ahli};
        \node[text=nvgray, font=\tiny, below=0.15cm of gpu] {ribuan tangan};
      \end{tikzpicture}
      \end{center}
    \end{column}
  \end{columns}
  \takeaway{FP16 bukan cuma hemat memori --- ia \textbf{menyalakan Tensor Core}, sehingga matmul jadi berkali lipat lebih cepat.}
\end{frame}

% ── Benchmark matmul speed ────────────────────────────────

        \item \textbf{warmup} $+$ sinkronisasi CUDA wajib agar angkanya jujur

% ── Tangga precision ──────────────────────────────────────
\begin{frame}{Tangga Precision: FP32 $\to$ FP16 $\to$ FP8}
  \framesubtitle{Makin sedikit bit $\to$ makin kecil memori \& makin cepat --- dengan ongkos ketelitian}
  \begin{center}
  \renewcommand{\arraystretch}{1.25}{\footnotesize\setlength{\tabcolsep}{5pt}
  \begin{tabular}{@{}p{0.14\linewidth}p{0.06\linewidth}p{0.13\linewidth}p{0.22\linewidth}p{0.34\linewidth}@{}}
    \toprule \rowcolor{nvcard!50}
    \textbf{Precision} & \textbf{Bit} & \textbf{Memori} & \textbf{Tensor Core} & \textbf{Catatan} \\ \midrule
    \textbf{FP32} & 32 & 1.0$\times$ & tidak & baseline riset \\
    \textbf{FP16} & 16 & \textbf{0.5$\times$} & ya (Volta+) & default deploy aman di T4 \\
    BF16 & 16 & 0.5$\times$ & ya (Ampere+) & \textbf{tidak optimal di T4} \\
    \textbf{FP8} & 8 & \textbf{0.25$\times$} & ya (Hopper/Ada+) & masa depan; tak jalan di T4 \\
    \bottomrule
  \end{tabular}}
  \end{center}
  \takeaway{Di T4 selalu \textbf{FP16, bukan BF16}. FP8 dipakai TensorRT-LLM di GPU baru --- di sini dipahami sebagai konsep.}
\end{frame}

% ── Benchmark memori + quantization ──────────────────────
\begin{frame}{Quantization 4-bit --- Memuat yang Tak Muat}
  \framesubtitle{Memori bobot Qwen2.5-1.5B di T4: tiga tingkat kompresi}
  \begin{columns}[T]
    \begin{column}{0.40\textwidth}
      \begin{itemize}\footnotesize
        \item FP32$\to$FP16: \textbf{2.0$\times$} hemat (6.18 $\to$ 3.10 GB)
        \item FP16$\to$4-bit: \textbf{2.7$\times$} lagi (3.09 $\to$ 1.16 GB)
        \item Gabungan $\sim$\textbf{5$\times$}: model 7B muat di T4 16 GB
        \item \textbf{NF4} + compute FP16 + double-quant
        \item Ini \emph{inference} quant (deploy) --- beda niat dari QLoRA \emph{training} (M04)
      \end{itemize}
    \end{column}
    \begin{column}{0.58\textwidth}
      \begin{center}
      \begin{tikzpicture}
        \begin{axis}[
          width=6.6cm, height=4.2cm,

  \takeaway{Quantization: penurunan kualitas kecil, penghematan memori besar --- itulah cara memuat model besar di GPU kecil.}
\end{frame}

% ── TensorRT vs TensorRT-LLM ─────────────────────────────
\begin{frame}{Compile: TensorRT vs TensorRT-LLM}
  \framesubtitle{Tuas ketiga --- ubah model jadi \emph{engine} native GPU}
  \begin{columns}[T]
    \begin{column}{0.49\textwidth}
      \textcolor{nvgreen}{\textbf{TensorRT}} --- compiler umum
      \begin{itemize}\footnotesize
        \item Model \emph{apa pun}: \texttt{model $\to$ ONNX $\to$ engine}
        \item Layer/kernel \textbf{fusion}, precision calibration, auto-tuning
        \item Engine terikat ke satu GPU
      \end{itemize}
    \end{column}
    \begin{column}{0.49\textwidth}
      \textcolor{nvorange}{\textbf{TensorRT-LLM}} --- khusus LLM
      \begin{itemize}\footnotesize
        \item \textbf{Paged KV-cache} --- tak ada VRAM terbuang
        \item \textbf{In-flight batching} --- GPU tak menganggur
        \item \textbf{FP8 / INT4} quant + fused attention
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}\textcolor{nvgray}{\scriptsize \texttt{trtllm-serve} mengangkat engine jadi server \textbf{OpenAI \texttt{/v1}} --- kontrak yang sama dengan Ollama (M04) \& NIM.}\end{center}
  \takeaway{TensorRT $=$ optimizer umum; \textbf{TensorRT-LLM} $=$ optimizer LLM dengan KV-cache \& batching sadar-LLM.}
\end{frame}
