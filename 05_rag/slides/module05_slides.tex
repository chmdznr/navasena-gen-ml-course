% ── Act 1: Apa itu RAG? ───────────────────────────────────
\acttitle{1}{Apa itu RAG?}{Memberi LLM akses ke pengetahuan eksternal}

% Slide 2 — Masalah LLM
\begin{frame}{Masalah LLM}
  \framesubtitle{Kenapa LLM saja tidak cukup?}
  \begin{itemize}
    \item LLM menjawab dari \textbf{ingatan} hasil pelatihannya saja
    \item[\textcolor{nvred}{$\bullet$}] Masalah 1: \textbf{halusinasi} --- mengarang jawaban meyakinkan tapi salah
    \item[\textcolor{nvred}{$\bullet$}] Masalah 2: \textbf{knowledge cutoff} --- tak tahu kejadian setelah training
    \item Contoh: tanya berita minggu lalu $\to$ LLM bisa salah / menebak
    \item Akibatnya: jawaban tak bisa dipercaya untuk fakta terbaru / spesifik
  \end{itemize}
  \vspace{0.5em}
  \begin{beamercolorbox}[rounded=true,shadow=false,sep=6pt]{block body}
    \centering\small\color{white}
    LLM pintar berbahasa, tapi \textbf{bukan sumber kebenaran fakta} --- ingatannya terbatas dan kadang ngawur.
  \end{beamercolorbox}
\end{frame}

% Slide 3 — Solusi RAG
\begin{frame}{Solusi: RAG}
  \framesubtitle{Retrieval-Augmented Generation}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{RAG} $=$ \textbf{Retrieval} $+$ \textbf{Generation}
        \item \textbf{Retrieval}: cari potongan info relevan dari dokumen (knowledge base)
        \item \textbf{Generation}: LLM menyusun jawaban dari \textbf{context} yang ditemukan
        \item Hasilnya: jawaban faktual, bisa di-update, halusinasi berkurang
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{center}
      \begin{tikzpicture}[scale=0.8, every node/.style={transform shape},
          b/.style={draw=nvgreen, fill=nvcard, rounded corners=5pt, text=white, font=\footnotesize, minimum width=3cm, minimum height=0.8cm, align=center}]
        \node[b] (q) {Pertanyaan};
        \node[b, below=0.5cm of q] (r) {Retrieval\\\scriptsize cari konteks};
        \node[b, below=0.5cm of r, draw=nvorange] (g) {Generation\\\scriptsize LLM jawab};
        \draw[->,thick,color=nvgreen] (q)--(r);
        \draw[->,thick,color=nvorange] (r)--(g);
      \end{tikzpicture}
      \end{center}
    \end{column}
  \end{columns}
  \vspace{0.3em}
  \begin{beamercolorbox}[rounded=true,shadow=false,sep=4pt]{block body}
    \centering\small\color{white}
    RAG tidak mengubah otak LLM --- ia \textbf{membekali} LLM dengan konteks faktual segar sebelum menjawab.
  \end{beamercolorbox}
\end{frame}

% Slide 4 — Analogi
\begin{frame}{Analogi: Ujian Buka Buku}
  \framesubtitle{LLM tidak menjawab dari ingatan saja}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textcolor{nvred}{\textbf{Tanpa RAG}} (tutup buku):
      \begin{itemize}
        \footnotesize
        \item Jawab dari ingatan $\to$ mudah lupa / salah
      \end{itemize}
      \vspace{0.4em}
      \textcolor{nvgreen}{\textbf{Dengan RAG}} (buka buku):
      \begin{itemize}
        \footnotesize
        \item Boleh lihat "buku contekan" dulu
        \item \textbf{Retrieval} $=$ buka halaman relevan dengan soal
        \item Konteks dibekalkan ke LLM sebelum menjawab
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{center}
      \resizebox{0.9\linewidth}{!}{%
      \begin{tikzpicture}
        \node[draw=nvgreen, fill=nvcard, rounded corners=8pt, text=white,

      }
      \end{center}

  \end{columns}
  \vspace{0.3em}
  \begin{center}\textcolor{nvlightgreen}{\small Dibekali konteks faktual dulu, baru menjawab --- bukan tebakan dari ingatan.}\end{center}
\end{frame}

\acttitle{2}{Komponen RAG}{Embedding, FAISS, knowledge base, dan generator}

% Slide 5 — Embeddings
\begin{frame}[fragile]{Embeddings \& SentenceTransformer}
  \framesubtitle{Mengubah teks menjadi angka yang bisa dibandingkan}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{Embedding}: teks $\to$ \textbf{vektor} (deretan angka)
        \item Kalimat \textbf{mirip makna} $\to$ vektor \textbf{berdekatan}
        \item Komputer paham jarak antar-angka, bukan kata
        \item Model \textbf{all-MiniLM-L6-v2} via \textbf{SentenceTransformer}
      \end{itemize}
      \begin{lstlisting}[language=Python, basicstyle=\ttfamily\scriptsize\color{nvwhite}]
embedder = SentenceTransformer(
  'sentence-transformers/all-MiniLM-L6-v2')
v = embedder.encode(["capital of France"])
      \end{lstlisting}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{center}
        \includegraphics[width=\linewidth]{figures/embedding_space.pdf}
      \end{center}
    \end{column}
  \end{columns}
\end{frame}

% Slide 6 — FAISS
\begin{frame}[fragile]{Similarity Search \& FAISS}
  \framesubtitle{Mencari vektor terdekat dengan cepat}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{FAISS} (Facebook AI Similarity Search): pencari vektor mirip super cepat
        \item \textbf{IndexFlatL2}: ukur \textbf{jarak Euclidean (L2)} antar vektor
        \item Makin kecil jarak $\to$ makin mirip maknanya
        \item Ambil \textbf{top-k} terdekat (notebook \texttt{k=1} untuk fokus; umumnya 3--5)
        \item Tetap cepat walau knowledge base ribuan dokumen
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{lstlisting}[language=Python, basicstyle=\ttfamily\scriptsize\color{nvwhite}]
index = faiss.IndexFlatL2(
            embedding_size)
distances, indices = index.search(
  query_embedding.astype('float32'), k)
      \end{lstlisting}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}\textcolor{nvlightgreen}{\small FAISS $=$ mesin pencari makna: dari query ke dokumen paling relevan dalam sekejap.}\end{center}
\end{frame}

% Slide 7 — Knowledge Base
\begin{frame}[fragile]{Knowledge Base}
  \framesubtitle{Sumber fakta yang dibaca sistem RAG}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{Knowledge base}: kumpulan dokumen / fakta sumber jawaban
        \item Contoh: Paris, Python, Machine Learning, Tembok China, dll
        \item Tiap dokumen di-\texttt{encode()} jadi vektor
        \item Semua vektor masuk \textbf{FAISS} via \texttt{index.add()}
        \item Tambah dokumen $=$ encode lalu masukkan (tanpa latih ulang)
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{lstlisting}[language=Python, basicstyle=\ttfamily\scriptsize\color{nvwhite}]
knowledge_base = [
  "...Paris...", "...Python...", ...]
embeddings = embedder.encode(
                knowledge_base)
index.add(embeddings.astype('float32'))
      \end{lstlisting}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}\textcolor{nvlightgreen}{\small Knowledge base $=$ ingatan eksternal LLM: cukup tambah dokumen, tak perlu latih ulang.}\end{center}
\end{frame}

% Slide 8 — Generator
\begin{frame}[fragile]{Generator LLM}
  \framesubtitle{TinyLlama menyusun jawaban akhir}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{Generator}: LLM penulis jawaban --- \textbf{TinyLlama} 1.1B Chat
        \item \texttt{torch.float32} (presisi penuh) agar \textbf{stabil di CPU}
        \item Dibungkus \texttt{pipeline("text-generation")}
        \item \texttt{temperature=0.3} $\to$ fokus; makin tinggi makin acak
        \item \texttt{top\_p=0.9} $\to$ \textbf{nucleus sampling}
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{lstlisting}[language=Python, basicstyle=\ttfamily\scriptsize\color{nvwhite}]
generator = pipeline(
  "text-generation",
  model=model, tokenizer=tokenizer,
  max_length=512, do_sample=True,
  temperature=0.3, top_p=0.9)
      \end{lstlisting}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}\textcolor{nvlightgreen}{\small TinyLlama float32 $+$ sampling rendah: jawaban fokus berbasis konteks, stabil tanpa GPU.}\end{center}
\end{frame}

% Slide 10 — Retrieve
\begin{frame}[fragile]{Langkah 1 --- Retrieve}
  \framesubtitle{Mencari konteks paling relevan}
  \begin{columns}[T]
    \begin{column}{0.44\textwidth}
      \begin{itemize}
        \footnotesize
        \item Query di-\textbf{embed} jadi \textbf{vektor}
        \item Dicocokkan ke \textbf{FAISS} (\texttt{IndexFlatL2})
        \item Ambil \textbf{top-k} terdekat (\texttt{k=1})
        \item Teks asli diambil dari \texttt{knowledge\_base}
      \end{itemize}
    \end{column}
    \begin{column}{0.56\textwidth}
      \begin{lstlisting}[language=Python, basicstyle=\ttfamily\tiny\color{nvwhite}]
def retrieve_relevant_context(query, k=1):
    q_emb = embedder.encode([query])
    dist, idx = index.search(
        q_emb.astype('float32'), k)
    docs = [knowledge_base[i] for i in idx[0]]
    return "\n".join(docs)
      \end{lstlisting}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}\textcolor{nvlightgreen}{\small Makin dekat letak vektor, makin mirip maknanya --- FAISS cari tetangga terdekat dengan cepat.}\end{center}
\end{frame}

% Slide 11 — Augment
\begin{frame}{Langkah 2 --- Augment}
  \framesubtitle{Menyisipkan konteks ke dalam prompt}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{Augment}: perkaya \textbf{prompt} dengan konteks hasil retrieve
        \item Konteks $+$ pertanyaan asli digabung jadi satu prompt
        \item Format: bagian \texttt{Context:} lalu \texttt{Question:}
        \item LLM punya bahan faktual, bukan sekadar menebak
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{center}
      \begin{tikzpicture}[scale=0.8, every node/.style={transform shape},

        \draw[->,thick,color=nvgreen] (c)--(p);
        \draw[->,thick,color=nvorange] (p)--(l);
      \end{tikzpicture}
      \end{center}
    \end{column}
  \end{columns}
  \vspace{0.3em}
  \begin{center}\textcolor{nvlightgreen}{\small Augment $=$ jembatan antara \textbf{retrieve} dan \textbf{generate}: konteks $+$ pertanyaan $\to$ satu prompt.}\end{center}
\end{frame}

% Slide 12 — Generate
\begin{frame}[fragile]{Langkah 3 --- Generate}
  \framesubtitle{LLM menjawab berbasis konteks}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \footnotesize
        \item \textbf{Generation}: LLM susun jawaban dari prompt yang diperkaya
        \item \textbf{TinyLlama} baca konteks $\to$ tulis jawaban relevan
        \item \texttt{float32} (bukan float16) agar stabil di CPU
        \item \texttt{temperature=0.3} $\to$ jawaban fokus/konsisten (kurang ngawur)
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{lstlisting}[language=Python, basicstyle=\ttfamily\scriptsize\color{nvwhite}]
response = generator(prompt,
  max_length=512, do_sample=True,
  temperature=0.3,
  num_return_sequences=1)
      \end{lstlisting}
    \end{column}
  \end{columns}
  \vspace{0.2em}
  \begin{center}\textcolor{nvlightgreen}{\small Berbekal konteks faktual, jawaban LLM lebih akurat \& tidak mengarang --- inti nilai RAG.}\end{center}
\end{frame}

\begin{frame}[fragile]{Kode Sistem RAG}
  \framesubtitle{Dua fungsi inti (disederhanakan): ambil konteks lalu susun jawaban}
  \begin{lstlisting}[language=Python, basicstyle=\ttfamily\scriptsize\color{nvwhite}]
def retrieve_relevant_context(query, k=1):
    q_emb = embedder.encode([query])
    dist, idx = index.search(q_emb.astype('float32'), k)
    return "\n".join(knowledge_base[i] for i in idx[0])

def generate_response(query, context):
    prompt = build_tinyllama_prompt(context, query)   # Context: ... Question: ...
    out = generator(prompt)[0]['generated_text']
    return clean_response(out)

# ask_question(): retrieve -> augment prompt -> generate
  \end{lstlisting}
  \vspace{0.1em}
  \begin{center}\textcolor{nvlightgreen}{\small Pola RAG: \textbf{retrieve} $\to$ \textbf{augment} prompt $\to$ \textbf{generate}. (Notebook asli pakai chat template TinyLlama.)}\end{center}
\end{frame}

% Slide 14 — Tips & Batasan
\begin{frame}{Tips \& Batasan}
  \framesubtitle{Kualitas jawaban bergantung pada beberapa pilihan}
  \begin{itemize}
    \item[\textcolor{nvgreen}{\checkmark}] \textbf{Ukuran chunk}: terlalu besar $\to$ konteks kabur; terlalu kecil $\to$ info penting hilang
    \item[\textcolor{nvgreen}{\checkmark}] \textbf{Nilai k}: berapa konteks diambil; notebook \texttt{k=1} (fokus), kadang butuh lebih
    \item[\textcolor{nvgreen}{\checkmark}] \textbf{Kualitas embedding}: model (all-MiniLM-L6-v2) menentukan ketepatan dokumen relevan
    \item[\textcolor{nvred}{$\bullet$}] RAG \textbf{bukan jaminan 100\% benar}: konteks kurang relevan $\to$ jawaban tetap meleset
  \end{itemize}
  \vspace{0.5em}
  \begin{beamercolorbox}[rounded=true,shadow=false,sep=6pt]{block body}
    \centering\small\color{white}
    RAG \textbf{mengurangi} halusinasi, bukan menghapusnya. Jawaban hanya sebaik konteks yang berhasil diambil.
  \end{beamercolorbox}
\end{frame}
